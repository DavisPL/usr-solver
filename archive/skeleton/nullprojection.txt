def parseStringVars(stringSet, unionFinder):
    stringDict = dict()
    returnVals = dict()
    truncate = dict()
    for elem in stringSet:
        stringDict[elem[0].name] = stringDict.get(elem[0].name, [])
        stringDict[elem[0].name].append(elem[1])
    for key, value in stringDict:
        retVal = None
        valueCopy = value.copy()
        lastPop = None
        while len(value) > 1:
            unionElems = set()
            newElementToAdd = None
            literalVal = None
            i = 0
            while i < len(value):
                expr = value[i]
				if expr is Concatenation:
                    unionElems.add(expr.left)
                    value[i] = expr.right
				else if expr is "":
                    value.pop(i)
                    lastPop = valueCopy.pop(i)
                    unionElems.add(expr)
                    truncate[key] = True
                    continue
                else:
					if expr is not StringVar:
                        unionElems.add(expr)
                    else:
                        value.pop(i)
                        lastPop = valueCopy.pop(i)
                        continue
                    value[i] = ""
                i+=1
            prev = None
            for elem in unionElems:
                if not prev:
                    prev = elem
                    continue
                else:
                    try:
                        unionFinder.union(prev, elem)
                        prev = elem
                    except Exception:
                        return EmptySet

        if len(valueCopy) > 0: 
            retVal = valueCopy[0]
        else:
            retVal = lastPop
        returnVals[key] = retVal
    return returnVals, truncate
