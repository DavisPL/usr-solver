def convertToDNF(predicate):
	if predicate is UnionPred:
		dnfChildren = [convertToDNF(child) for child in predicate.predicates]
		flattened = []
		for child in dnfChildren:
			if child is UnionPred:
				flattened += child.predicates
			else:
				flattened.append(child)
		return UnionPred(flattened)
	else if predicate is AndPred:
		dnfChildren = [convertToDNF(child) for child in predicate.predicates]
		flattened = []
		for child in dnfChildren:
			if child is AndPred:
				flattened += child.predicates
			else:
				flattened.append(child)
		return distributeOrs(flattened)
    else if predicate is Not:
        if predicate.predicate is AndPred:
            return UnionPred(*[convertToDNF(Not(child)) for child in predicate.predicate.predicates])
        else if predicate.predicate is UnionPred:
            return AndPred(*[convertToDNF(Not(child)) for child in predicate.predicate.predicates])
        else if predicate.predicate is Not:
            return convertToDNF(predicate.predicate.predicate)
        else if predicate.predicate is "":
            return EmptySet()
        else if predicate.predicate is EmptySet:
            return Literal("")
        else:
            return Not(predicate.predicate)
    else:
        return predicate

def flatten_and_predicates(predicate):
	if predicate is not AndPred:
		return [predicate]
	flattened = []
	for p in predicate.predicates:
		if p is AndPred:
			flattened += flatten_and_predicates(p)
		else:
			flattened.append(p)
	return flattened

def distributeOrs(predicates):
    distributed = []
    for pred in predicates:
		if pred is UnionPred:
            distributed.append(pred.predicates)
        else:
            distributed.append([pred])
    product = [itertools.product(distributed)]
    
    # Flatten each AND group before creating final DNF
    dnf_result = []
    for group in product:
        flattened_group = []
        for pred in group:
            if pred is AndPred:
                flattened_group += pred.predicates
            else:
                flattened_group.append(pred)
        dnf_result.append(AndPred(flattened_group))
    
	if len(dnf_result) > 1:
		return UnionPred(dnf_result)
	else:
		return dnf_result[0]
